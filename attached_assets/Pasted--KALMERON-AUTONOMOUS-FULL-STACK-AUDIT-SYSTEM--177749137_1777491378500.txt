```typescript
export type Severity = 'critical' | 'high' | 'medium' | 'low' | 'info';
export type Category =
  | 'code-quality' | 'security' | 'authentication'
  | 'frontend' | 'backend' | 'storage'
  | 'performance' | 'seo' | 'ai-agents' | 'business';

export interface AuditFinding {
  id: string;
  category: Category;
  severity: Severity;
  title: string;
  description: string;
  location?: string;       // مسار الملف أو URL
  evidence?: string;       // دليل المشكلة
  fix?: string;            // اقتراح الإصلاح
  autoFixable: boolean;
  references?: string[];   // روابط OWASP أو Next.js docs
}

export interface ModuleResult {
  module: string;
  category: Category;
  score: number;           // 0-100
  findings: AuditFinding[];
  duration: number;        // ms
  passed: number;
  failed: number;
}

export interface AuditReport {
  id: string;
  timestamp: string;
  projectName: string;
  overallScore: number;    // 0-100
  grade: 'A' | 'B' | 'C' | 'D' | 'F';
  modules: ModuleResult[];
  criticalCount: number;
  autoFixCount: number;
  manualFixCount: number;
  topPriorities: AuditFinding[];  // أهم 10 مشاكل
  summary: string;
}
```

```typescript
// audit/modules/01-code-quality.ts
import { execSync } from 'child_process';
import { AuditFinding } from '../types';

export async function auditCodeQuality(): Promise<AuditFinding[]> {
  const findings: AuditFinding[] = [];

  // ── 1. TypeScript strict check ──
  try {
    execSync(
      'node --stack-size=8192 ./node_modules/typescript/bin/tsc --noEmit 2>&1',
      { encoding: 'utf8' }
    );
  } catch (e: any) {
    const errorCount = (e.stdout?.match(/error TS/g) || []).length;
    if (errorCount > 0) {
      findings.push({
        id: 'CQ-001',
        category: 'code-quality',
        severity: 'high',
        title: `${errorCount} خطأ TypeScript`,
        description: 'الكود لا يمر من فحص TypeScript الصارم',
        evidence: e.stdout?.slice(0, 500),
        fix: 'شغّل: node --stack-size=8192 ./node_modules/typescript/bin/tsc --noEmit وأصلح كل خطأ',
        autoFixable: false,
        references: ['https://www.typescriptlang.org/docs/handbook/2/types-from-types.html'],
      });
    }
  }

  // ── 2. ESLint errors ──
  try {
    const result = execSync('npm run lint -- --format json 2>&1', { encoding: 'utf8' });
    const lintData = JSON.parse(result);
    const errors = lintData.flatMap((f: any) => f.messages.filter((m: any) => m.severity === 2));
    if (errors.length > 0) {
      findings.push({
        id: 'CQ-002',
        category: 'code-quality',
        severity: 'medium',
        title: `${errors.length} خطأ ESLint`,
        description: 'مخالفات قواعد الكود',
        fix: 'شغّل: npm run lint -- --fix لإصلاح التلقائي',
        autoFixable: true,
        references: ['https://eslint.org/docs/latest/'],
      });
    }
  } catch { /* lint failed entirely */ }

  // ── 3. Dependency vulnerabilities ──
  try {
    const audit = JSON.parse(
      execSync('npm audit --json 2>/dev/null', { encoding: 'utf8' })
    );
    const highCrit = (audit.metadata?.vulnerabilities?.high ?? 0) +
                     (audit.metadata?.vulnerabilities?.critical ?? 0);
    if (highCrit > 0) {
      findings.push({
        id: 'CQ-003',
        category: 'code-quality',
        severity: highCrit > 5 ? 'critical' : 'high',
        title: `${highCrit} ثغرة في المكتبات (High/Critical)`,
        description: 'مكتبات تحتوي على ثغرات أمنية معروفة',
        fix: 'شغّل: npm audit fix أو حدّث المكتبات المتأثرة يدوياً',
        autoFixable: true,
        references: ['https://docs.npmjs.com/auditing-package-dependencies-for-security-vulnerabilities'],
      });
    }
  } catch { /* audit parse failed */ }

  // ── 4. Unused dependencies check ──
  try {
    const pkg = JSON.parse(
      require('fs').readFileSync('package.json', 'utf8')
    );
    const depCount = Object.keys(pkg.dependencies || {}).length;
    if (depCount > 80) {
      findings.push({
        id: 'CQ-004',
        category: 'code-quality',
        severity: 'low',
        title: `${depCount} dependency — قد تكون بعضها غير مستخدمة`,
        description: 'عدد كبير من المكتبات قد يزيد حجم الـ bundle',
        fix: 'راجع: npx depcheck لإيجاد المكتبات غير المستخدمة',
        autoFixable: false,
      });
    }
  } catch { /* */ }

  // ── 5. Console.log في production code ──
  try {
    const result = execSync(
      'grep -rn "console\\.log" app/ src/ --include="*.ts" --include="*.tsx" | grep -v ".test." | grep -v "// " | wc -l',
      { encoding: 'utf8' }
    ).trim();
    const count = parseInt(result, 10);
    if (count > 10) {
      findings.push({
        id: 'CQ-005',
        category: 'code-quality',
        severity: 'low',
        title: `${count} console.log في الكود الإنتاجي`,
        description: 'console.log تُبطئ الأداء وقد تكشف بيانات حساسة',
        fix: 'استبدل بـ pino logger أو احذف الـ console.log غير الضرورية',
        autoFixable: false,
      });
    }
  } catch { /* */ }

  // ── 6. TODO/FIXME comments ──
  try {
    const result = execSync(
      'grep -rn "TODO:\\|FIXME:\\|HACK:\\|XXX:" app/ src/ --include="*.ts" --include="*.tsx" | wc -l',
      { encoding: 'utf8' }
    ).trim();
    const count = parseInt(result, 10);
    if (count > 5) {
      findings.push({
        id: 'CQ-006',
        category: 'code-quality',
        severity: 'info',
        title: `${count} تعليق TODO/FIXME في الكود`,
        description: 'توجد مهام معلقة في الكود',
        fix: 'راجع وأكمل أو احذف التعليقات المعلقة قبل الإطلاق',
        autoFixable: false,
      });
    }
  } catch { /* */ }

  return findings;
}
```

```typescript
// audit/modules/02-security.ts
// مبني على: OWASP Top 10, Next.js Security Docs, Arcjet Security Checklist 2025
import { execSync } from 'child_process';
import { readFileSync, existsSync } from 'fs';
import { AuditFinding } from '../types';

export async function auditSecurity(): Promise<AuditFinding[]> {
  const findings: AuditFinding[] = [];

  // ── OWASP A02: Cryptographic Failures — Secrets في الكود ──
  const secretPatterns = [
    { pattern: /AAAA[A-Za-z0-9_-]{20,}/, name: 'Firebase token محتمل' },
    { pattern: /sk_live_[A-Za-z0-9]{24}/, name: 'Stripe Secret Key' },
    { pattern: /sk_test_[A-Za-z0-9]{24}/, name: 'Stripe Test Key' },
    { pattern: /AIzaSy[A-Za-z0-9_-]{33}/, name: 'Google API Key' },
    { pattern: /ghp_[A-Za-z0-9]{36}/, name: 'GitHub Token' },
    { pattern: /"private_key":\s*"-----BEGIN/, name: 'Private Key مُضمَّن' },
  ];

  const filesToScan = ['app/', 'src/', 'components/', '.env.local'].filter(existsSync);

  for (const { pattern, name } of secretPatterns) {
    try {
      const result = execSync(
        `grep -rn "${pattern.source}" ${filesToScan.join(' ')} --include="*.ts" --include="*.tsx" --include="*.js" 2>/dev/null | grep -v ".env" | grep -v "example" | wc -l`,
        { encoding: 'utf8', stdio: 'pipe' }
      ).trim();

      if (parseInt(result, 10) > 0) {
        findings.push({
          id: `SEC-001-${name.replace(/\s/g, '')}`,
          category: 'security',
          severity: 'critical',
          title: `🔴 Secret محتمل في الكود: ${name}`,
          description: 'مفتاح سري مُضمَّن في الكود — خطر كبير جداً',
          fix: 'انقل الـ secret لـ .env.local فوراً وأبلّغ عنه كـ compromised',
          autoFixable: false,
          references: ['https://owasp.org/Top10/A02_2021-Cryptographic_Failures/'],
        });
      }
    } catch { /* */ }
  }

  // ── OWASP A05: Security Headers ──
  try {
    const nextConfig = readFileSync('next.config.ts', 'utf8');

    const requiredHeaders = [
      { header: 'Content-Security-Policy', id: 'SEC-CSP' },
      { header: 'X-Frame-Options', id: 'SEC-XFO' },
      { header: 'X-Content-Type-Options', id: 'SEC-CTO' },
      { header: 'Strict-Transport-Security', id: 'SEC-HSTS' },
      { header: 'Permissions-Policy', id: 'SEC-PP' },
    ];

    for (const { header, id } of requiredHeaders) {
      if (!nextConfig.includes(header)) {
        findings.push({
          id,
          category: 'security',
          severity: 'high',
          title: `Header أمني ناقص: ${header}`,
          description: `Security header ${header} غير موجود في next.config.ts`,
          location: 'next.config.ts',
          fix: `أضف ${header} في دالة headers() داخل next.config.ts`,
          autoFixable: true,
          references: ['https://owasp.org/www-project-secure-headers/'],
        });
      }
    }

    // تحقق من HSTS max-age
    if (nextConfig.includes('Strict-Transport-Security')) {
      const hstsMatch = nextConfig.match(/max-age=(\d+)/);
      if (hstsMatch && parseInt(hstsMatch[1], 10) < 15552000) {
        findings.push({
          id: 'SEC-HSTS-WEAK',
          category: 'security',
          severity: 'medium',
          title: 'HSTS max-age قصير جداً',
          description: `max-age=${hstsMatch[1]} أقل من 180 يوم (15,552,000 ثانية)`,
          location: 'next.config.ts',
          fix: 'اضبط max-age=63072000 (سنتان) مع includeSubDomains; preload',
          autoFixable: true,
        });
      }
    }
  } catch { /* */ }

  // ── OWASP A03: Injection — XSS via dangerouslySetInnerHTML ──
  try {
    const dangerousUsages = execSync(
      'grep -rn "dangerouslySetInnerHTML" app/ src/ components/ --include="*.tsx" 2>/dev/null',
      { encoding: 'utf8' }
    ).trim().split('\n').filter(Boolean);

    for (const usage of dangerousUsages) {
      if (!usage.includes('DOMPurify') && !usage.includes('sanitize')) {
        findings.push({
          id: 'SEC-XSS-001',
          category: 'security',
          severity: 'high',
          title: 'dangerouslySetInnerHTML بدون sanitization',
          description: 'استخدام dangerouslySetInnerHTML بدون تنظيف المحتوى — خطر XSS',
          location: usage.split(':')[0],
          evidence: usage,
          fix: 'استخدم DOMPurify.sanitize() قبل تمرير المحتوى لـ dangerouslySetInnerHTML',
          autoFixable: false,
          references: ['https://owasp.org/www-community/attacks/xss/'],
        });
      }
    }
  } catch { /* */ }

  // ── Rate Limiting فحص ──
  try {
    const rateLimitUsage = parseInt(
      execSync(
        'grep -rn "rateLimit\\|rateLimiter\\|rate_limit" app/api/ --include="*.ts" 2>/dev/null | wc -l',
        { encoding: 'utf8' }
      ).trim(), 10
    );
    const apiRoutes = parseInt(
      execSync('find app/api -name "route.ts" | wc -l', { encoding: 'utf8' }).trim(), 10
    );

    if (rateLimitUsage < apiRoutes * 0.5) {
      findings.push({
        id: 'SEC-RATE-001',
        category: 'security',
        severity: 'high',
        title: `Rate limiting ناقص — ${apiRoutes} route و${rateLimitUsage} فحص فقط`,
        description: 'أكثر من نصف API routes بدون rate limiting',
        fix: 'أضف middleware لـ rate limiting على كل routes في proxy.ts',
        autoFixable: false,
        references: ['https://owasp.org/API-Security/editions/2023/en/0xa4-unrestricted-resource-consumption/'],
      });
    }
  } catch { /* */ }

  // ── CVE-2025-29927: Next.js Middleware Bypass ──
  try {
    const pkgJson = JSON.parse(readFileSync('package.json', 'utf8'));
    const nextVersion = pkgJson.dependencies?.next || '';
    const versionMatch = nextVersion.match(/(\d+)\.(\d+)\.(\d+)/);
    if (versionMatch) {
      const [, major, minor, patch] = versionMatch.map(Number);
      // الإصدارات المتأثرة < 15.2.3 أو < 14.2.25 أو < 13.5.9
      const isVulnerable =
        (major === 15 && (minor < 2 || (minor === 2 && patch < 3))) ||
        (major === 14 && (minor < 2 || (minor === 2 && patch < 25))) ||
        major < 13;

      if (isVulnerable) {
        findings.push({
          id: 'SEC-CVE-2025-29927',
          category: 'security',
          severity: 'critical',
          title: 'CVE-2025-29927: ثغرة تجاوز Middleware في Next.js',
          description: `إصدار Next.js الحالي (${nextVersion}) عُرضة لثغرة تجاوز المصادقة`,
          location: 'package.json',
          fix: 'حدّث Next.js لأحدث إصدار: npm install next@latest',
          autoFixable: true,
          references: ['https://nextjs.org/blog/cve-2025-29927'],
        });
      }
    }
  } catch { /* */ }

  // ── CSRF Protection ──
  try {
    const serverActions = execSync(
      'grep -rn "use server" app/ --include="*.ts" --include="*.tsx" | wc -l',
      { encoding: 'utf8' }
    ).trim();
    const csrfProtection = execSync(
      'grep -rn "csrf\\|CSRF\\|x-csrf" app/ src/ --include="*.ts" | wc -l',
      { encoding: 'utf8' }
    ).trim();

    if (parseInt(serverActions) > 0 && parseInt(csrfProtection) === 0) {
      findings.push({
        id: 'SEC-CSRF-001',
        category: 'security',
        severity: 'medium',
        title: 'Server Actions بدون CSRF protection صريح',
        description: 'Next.js يوفر حماية ضمنية، لكن يُنصح بـ custom validation',
        fix: 'تحقق من Origin header في كل Server Action: https://nextjs.org/docs/app/building-your-application/data-fetching/server-actions-and-mutations#security',
        autoFixable: false,
        references: ['https://owasp.org/www-community/attacks/csrf'],
      });
    }
  } catch { /* */ }

  return findings;
}
```

```typescript
// audit/modules/03-authentication.ts
import { readFileSync, existsSync } from 'fs';
import { execSync } from 'child_process';
import { AuditFinding } from '../types';

export async function auditAuthentication(): Promise<AuditFinding[]> {
  const findings: AuditFinding[] = [];

  // ── تحقق Firebase Admin SDK صحيح ──
  const adminFiles = ['src/lib/firebase-admin.ts', 'lib/firebase-admin.ts'];
  const adminFile = adminFiles.find(existsSync);

  if (!adminFile) {
    findings.push({
      id: 'AUTH-001',
      category: 'authentication',
      severity: 'critical',
      title: 'Firebase Admin SDK غير موجود',
      description: 'ملف firebase-admin.ts مفقود — المصادقة server-side لن تعمل',
      fix: 'أنشئ src/lib/firebase-admin.ts مع initializeApp صحيح',
      autoFixable: false,
    });
  } else {
    const content = readFileSync(adminFile, 'utf8');

    if (!content.includes('getApps()') && !content.includes('getApp(')) {
      findings.push({
        id: 'AUTH-002',
        category: 'authentication',
        severity: 'high',
        title: 'Firebase Admin قد يُهيَّأ مرات متعددة',
        description: 'بدون فحص getApps() قد ينهار التطبيق عند إعادة التحميل',
        location: adminFile,
        fix: 'استخدم: if (!getApps().length) { initializeApp(...) }',
        autoFixable: true,
      });
    }

    if (!content.includes('FIREBASE_ADMIN_PRIVATE_KEY') &&
        !content.includes('process.env')) {
      findings.push({
        id: 'AUTH-003',
        category: 'authentication',
        severity: 'critical',
        title: 'Firebase Admin credentials hard-coded',
        description: 'بيانات الاعتماد مُضمَّنة في الكود بدلاً من env vars',
        location: adminFile,
        fix: 'استخدم process.env.FIREBASE_ADMIN_PRIVATE_KEY',
        autoFixable: false,
      });
    }
  }

  // ── فحص AuthContext ──
  const authContextFiles = [
    'contexts/AuthContext.tsx',
    'src/contexts/AuthContext.tsx',
    'context/AuthContext.tsx',
  ];
  const authContext = authContextFiles.find(existsSync);

  if (!authContext) {
    findings.push({
      id: 'AUTH-004',
      category: 'authentication',
      severity: 'critical',
      title: 'AuthContext غير موجود',
      description: 'لا يوجد Context لإدارة حالة المصادقة',
      fix: 'أنشئ contexts/AuthContext.tsx',
      autoFixable: false,
    });
  }

  // ── فحص المسارات المحمية في proxy.ts ──
  const proxyFile = existsSync('proxy.ts') ? 'proxy.ts' : null;
  if (proxyFile) {
    const proxyContent = readFileSync(proxyFile, 'utf8');
    const criticalPaths = ['/dashboard', '/admin', '/billing', '/settings'];

    for (const path of criticalPaths) {
      if (!proxyContent.includes(path)) {
        findings.push({
          id: `AUTH-005-${path.replace('/', '')}`,
          category: 'authentication',
          severity: 'high',
          title: `المسار "${path}" قد لا يكون محمياً في proxy.ts`,
          description: `لم يُعثر على "${path}" في قائمة المسارات المحمية`,
          location: 'proxy.ts',
          fix: `أضف "${path}" لقائمة PROTECTED_PATHS في proxy.ts`,
          autoFixable: true,
        });
      }
    }
  } else {
    findings.push({
      id: 'AUTH-006',
      category: 'authentication',
      severity: 'critical',
      title: 'proxy.ts غير موجود',
      description: 'لا يوجد Edge Proxy لحماية المسارات',
      fix: 'أنشئ proxy.ts مع AuthGuard middleware',
      autoFixable: false,
    });
  }

  // ── Session Security ──
  try {
    const authContent = authContext ? readFileSync(authContext, 'utf8') : '';
    if (authContent.includes('localStorage') && !authContent.includes('httpOnly')) {
      findings.push({
        id: 'AUTH-007',
        category: 'authentication',
        severity: 'medium',
        title: 'Session token في localStorage',
        description: 'تخزين tokens في localStorage عُرضة لـ XSS attacks',
        fix: 'استخدم httpOnly cookies أو Firebase Auth onAuthStateChanged',
        autoFixable: false,
        references: ['https://owasp.org/www-community/HttpOnly'],
      });
    }
  } catch { /* */ }

  // ── Token Verification في API Routes ──
  try {
    const apiRoutesCount = parseInt(
      execSync('find app/api -name "route.ts" | wc -l', { encoding: 'utf8' }).trim(), 10
    );
    const verifyCount = parseInt(
      execSync(
        'grep -rn "verifyIdToken\\|getAuthenticatedUser\\|withAuth" app/api/ --include="*.ts" 2>/dev/null | wc -l',
        { encoding: 'utf8' }
      ).trim(), 10
    );

    if (verifyCount < apiRoutesCount * 0.6) {
      findings.push({
        id: 'AUTH-008',
        category: 'authentication',
        severity: 'high',
        title: `${apiRoutesCount - verifyCount} API route بدون تحقق من الهوية`,
        description: 'كثير من API routes لا تتحقق من Firebase ID Token',
        fix: 'أضف getAuthenticatedUser() أو withAuth() middleware لكل route محمي',
        autoFixable: false,
      });
    }
  } catch { /* */ }

  return findings;
}
```

```typescript
// audit/modules/04-frontend.ts
import { execSync } from 'child_process';
import { readFileSync, existsSync } from 'fs';
import { AuditFinding } from '../types';

export async function auditFrontend(): Promise<AuditFinding[]> {
  const findings: AuditFinding[] = [];

  // ── RTL Consistency ──
  try {
    const layoutFile = 'app/layout.tsx';
    if (existsSync(layoutFile)) {
      const layout = readFileSync(layoutFile, 'utf8');
      if (!layout.includes('dir="rtl"') && !layout.includes("dir='rtl'")) {
        findings.push({
          id: 'FE-001',
          category: 'frontend',
```

```typescript
import { AuditFinding } from '../types';

export async function auditStorage(): Promise<AuditFinding[]> {
  const findings: AuditFinding[] = [];

  // ── Firestore Rules ──
  if (!existsSync('firestore.rules')) {
    findings.push({
      id: 'STOR-001',
      category: 'storage',
      severity: 'critical',
      title: 'firestore.rules ملف مفقود',
      description: 'بدون rules: Firestore قد يكون مفتوحاً للعموم',
      fix: 'أنشئ firestore.rules مع deny-all default',
      autoFixable: false,
    });
    return findings;
  }

  const rules = readFileSync('firestore.rules', 'utf8');

  // تحقق من deny-all default
  if (!rules.includes('allow read, write: if false') &&
      !rules.includes('allow read,write: if false')) {
    findings.push({
      id: 'STOR-002',
      category: 'storage',
      severity: 'critical',
      title: 'Firestore: لا يوجد deny-all default rule',
      description: 'بدون rule افتراضي لـ deny، قد تكون بعض collections مكشوفة',
      location: 'firestore.rules',
      fix: 'أضف في النهاية: match /{document=**} { allow read, write: if false; }',
      autoFixable: true,
    });
  }

  // تحقق من allow write: if true (خطر كبير)
  const openWrites = (rules.match(/allow write: if true/g) || []).length;
  if (openWrites > 0) {
    findings.push({
      id: 'STOR-003',
      category: 'storage',
      severity: 'critical',
      title: `${openWrites} collection مفتوح للكتابة للجميع`,
      description: '"allow write: if true" يسمح لأي شخص بالكتابة',
      location: 'firestore.rules',
      fix: 'استبدل "if true" بـ "if request.auth != null" أو rule أكثر تحديداً',
      autoFixable: false,
    });
  }

  // تحقق من user_credits — يجب أن يكون Admin SDK فقط
  if (rules.includes('user_credits')) {
    const creditSection = rules.split('user_credits')[1]?.slice(0, 200) || '';
    if (creditSection.includes('allow write') && !creditSection.includes('if false')) {
      findings.push({
        id: 'STOR-004',
        category: 'storage',
        severity: 'critical',
        title: 'user_credits قابل للتعديل من العميل',
        description: 'المستخدم يمكنه تعديل رصيده بنفسه — خطر احتيال',
        location: 'firestore.rules',
        fix: 'اضبط user_credits: allow read: if isOwner(); allow write: if false;',
        autoFixable: true,
      });
    }
  }

  // Backup strategy
  const hasBackupCron = existsSync('app/api/cron/firestore-backup/route.ts') ||
    readFileSync('vercel.json', 'utf8').includes('firestore-backup');
  if (!hasBackupCron) {
    findings.push({
      id: 'STOR-005',
      category: 'storage',
      severity: 'medium',
      title: 'لا يوجد Firestore backup مجدول',
      description: 'فقدان البيانات ممكن بدون نسخ احتياطية منتظمة',
      fix: 'أضف cron job يشغّل Firestore export يومياً',
      autoFixable: false,
    });
  }

  // Firestore indexes
  if (!existsSync('firestore.indexes.json')) {
    findings.push({
      id: 'STOR-006',
      category: 'storage',
      severity: 'low',
      title: 'firestore.indexes.json مفقود',
      description: 'queries معقدة قد تفشل بدون composite indexes',
      fix: 'أنشئ firestore.indexes.json مع الـ indexes اللازمة',
      autoFixable: false,
    });
  }

  return findings;
}
```

```typescript
import { execSync } from 'child_process';
import { readFileSync, existsSync } from 'fs';
import { AuditFinding } from '../types';

export async function auditPerformance(): Promise<AuditFinding[]> {
  const findings: AuditFinding[] = [];
  const BASE_URL = process.env.QA_BASE_URL ?? 'http://localhost:5000';

  // ── Bundle Size Check ──
  if (existsSync('.next/analyze')) {

      title: 'Bundle analyzer نتائج موجودة',

      fix: 'شغّل: ANALYZE=true npm run build لرؤية التقرير',
      autoFixable: false,
    });
  }

  // ── Lighthouse Score ──
  try {
    const lighthouseResult = execSync(
      `npx lighthouse ${BASE_URL} --only-categories=performance,accessibility,seo,best-practices --output=json --quiet --chrome-flags="--headless --no-sandbox" 2>/dev/null`,
      { encoding: 'utf8', timeout: 120000 }
    );
    const report = JSON.parse(lighthouseResult);
    const categories = report.categories;

    const checks = [
      { key: 'performance', name: 'الأداء', threshold: 0.7 },
      { key: 'accessibility', name: 'إمكانية الوصول', threshold: 0.8 },
      { key: 'seo', name: 'SEO', threshold: 0.8 },
      { key: 'best-practices', name: 'أفضل الممارسات', threshold: 0.8 },
    ];

    for (const check of checks) {
      const score = categories[check.key]?.score ?? 0;
      if (score < check.threshold) {
        findings.push({
          id: `PERF-LH-${check.key.toUpperCase()}`,
          category: 'performance',
          severity: score < 0.5 ? 'high' : 'medium',
          title: `Lighthouse ${check.name}: ${Math.round(score * 100)}/100`,
          description: `نقاط Lighthouse ${check.name} أقل من ${check.threshold * 100}`,
          evidence: `Score: ${Math.round(score * 100)}/100 (الهدف: ${check.threshold * 100})`,

          autoFixable: false,
          references: ['https://developer.chrome.com/docs/lighthouse/'],
        });
      }
    }

    // Core Web Vitals
    const audits = report.audits;
    if (audits['largest-contentful-paint']?.numericValue > 2500) {

        autoFixable: false,
        references: ['https://web.dev/lcp/'],
      });
    }

    if (audits['cumulative-layout-shift']?.numericValue > 0.1) {
      findings.push({
        id: 'PERF-CLS',
        category: 'performance',
        severity: 'medium',
        title: `CLS مرتفع: ${audits['cumulative-layout-shift'].numericValue.toFixed(3)}`,
        description: 'Cumulative Layout Shift يجب أن يكون < 0.1',
        fix: 'أضف أبعاد صريحة للصور والـ dynamic content',
        autoFixable: false,
        references: ['https://web.dev/cls/'],
      });
    }
  } catch {
    findings.push({
      id: 'PERF-LH-UNAVAILABLE',
      category: 'performance',
      severity: 'info',
      title: 'Lighthouse غير متاح في هذه البيئة',

      });
    }
  } catch { /* */ }
```

```typescript
// audit/scorer.ts
import { AuditFinding, AuditReport, ModuleResult } from './types';

export function calculateScore(findings: AuditFinding[]): number {
  const weights = { critical: 25, high: 12, medium: 5, low: 2, info: 0 };
  const maxPenalty = 100;

  const penalty = findings
    .filter(f => ['critical', 'high', 'medium', 'low'].includes(f.severity))
    .reduce((sum, f) => sum + (weights[f.severity] ?? 0), 0);

  return Math.max(0, Math.round(100 - Math.min(penalty, maxPenalty)));
}

export function getGrade(score: number): 'A' | 'B' | 'C' | 'D' | 'F' {
  if (score >= 90) return 'A';
  if (score >= 80) return 'B';
  if (score >= 65) return 'C';
  if (score >= 50) return 'D';
  return 'F';
}
```

```typescript
  if (report.criticalCount > 0) process.exit(1);
  return report;
}

// نقطة الدخول
runFullAudit().catch(err => {
  console.error('فشل نظام الفحص:', err);
  process.exit(1);
});
```

```yaml
name: 🔍 Kalmeron Full Audit

on:

    .join('\n') : '### ✅ لا توجد مشاكل حرجة!'}
`;
            
            github.rest.issues.createComment({
              issue_number: context.issue.number,
              owner: context.repo.owner,
              repo: context.repo.repo,
              body
            });
```
